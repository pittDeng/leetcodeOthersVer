class Solution {
public:
    string getPermutation(int n, int k) {
        vector<int>factorial(n,1);
        vector<int>nums(n);
        for(int i=0;i<n;)nums[i]=++i;
        for(int i=2;i<n;++i)factorial[i]=factorial[i-1]*i;
        k-=1;
        string res="";
        for(int i=0;i<n;++i)
        {
            res+=('0'+nums[k/factorial[n-i-1]]);
            nums.erase(nums.begin()+k/factorial[n-i-1]);
            k=k%factorial[n-i-1];
        }
        return res;
    }
};
